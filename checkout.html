async function processarFluxo() { const status = document.getElementById('status'); const btn = document.getElementById('btn'); const email = document.getElementById('email').value; const cpf = document.getElementById('cpf').value.replace(/\D/g, ''); if (!email || !cpf) { status.innerHTML = `Preencha e-mail e CPF!`; return; } status.innerHTML = "Verificando se você já possui acesso..."; btn.disabled = true; try { // 1. CONFERE GLOBALMENTE SE ESSE CPF/EMAIL JÁ PAGOU const conferência = await call('/api/checkout', { action: 'check_global', payload: { cpfCnpj: cpf, email: email } }); if (conferência.data && conferência.data.length > 0) { redirecionar(); return; } // 2. Se não pagou, segue para criar cliente e pagamento status.innerHTML = "Identificando cadastro..."; const cli = await call('/api/checkout', { action: 'create_customer', payload: { name: document.getElementById('nome').value, email: email, cpfCnpj: cpf } }); status.innerHTML = "Gerando seu QR Code..."; const pag = await call('/api/checkout', { action: 'create_payment', payload: { customer: cli.id, billingType: 'PIX', value: 35.00, dueDate: new Date(Date.now() + 86400000).toISOString().split('T')[0] } }); const qr = await call('/api/checkout', { action: 'pix_qrcode', payload: { paymentId: pag.id } }); document.getElementById('form-container').style.display = 'none'; status.innerHTML = `
${qr.payload}

Aguardando pagamento...

`; iniciarVigia(pag.id); } catch (err) { status.innerHTML = `Erro: ${err.message}`; btn.disabled = false; } }
