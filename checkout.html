async function gerarPix() { const status = document.getElementById('status'); const btn = document.getElementById('btn'); status.innerHTML = "Verificando cadastro..."; btn.disabled = true; const dados = { name: document.getElementById('nome').value, email: document.getElementById('email').value, cpfCnpj: document.getElementById('cpf').value.replace(/\D/g, '') }; try { // 1. Localiza ou cria o cliente const cli = await callAPI('create_customer', dados); // 2. CONFERE SE JÁ PAGOU ANTES status.innerHTML = "Conferindo pagamentos anteriores..."; const existing = await callAPI('check_existing', { customerId: cli.id }); if (existing.data && existing.data.length > 0) { status.innerHTML = `
Você já possui um pagamento confirmado! Redirecionando...
`; setTimeout(() => { window.location.href = "https://vocal-divine-flow.lovable.app/obrigado"; }, 2000); return; } // 3. Se não pagou, segue o fluxo normal de gerar PIX status.innerHTML = "Gerando seu PIX..."; const pag = await callAPI('create_payment', { customer: cli.id, billingType: 'PIX', value: 35.00, dueDate: new Date(Date.now() + 86400000).toISOString().split('T')[0] }); const qr = await callAPI('pix_qrcode', { paymentId: pag.id }); document.getElementById('form-container').style.display = 'none'; status.innerHTML = `
${qr.payload}

Aguardando pagamento...

`; iniciarVerificacao(pag.id); } catch (err) { status.innerHTML = `
⚠️ ${err.message}
`; btn.disabled = false; } }
